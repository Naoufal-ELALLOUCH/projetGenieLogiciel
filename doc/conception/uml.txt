package "fr.enseirb.t2.it340.modeles" {
	class Utilisateur {
		int idUtilisateur
		String email
		String password
	}

	class Laboratoire {
		int idLaboratoire
		String nom
	}

	class Atelier {
		int idAtelier
		String titre
		Set<String> themes
		String zone
		Set<String> orateurs
		Set<String> partenaires
		Set<String> cibles
		String remarques

		void ajoutCreneau(Date date, int capacite)
		void modifierCreneau(int idCreneau, Date date, int capacite)
		void supprimerCreneau(int idCreneau)
	}

	enum Statut {
		PROPOSE
		VALIDE
		CLOTURE
	}

	class Creneau {
		int idCreneau
		int capacite
		Date date
		Date getDate()
		void setDate(Date date)
		int getCapacite()
		void setCapacite(int capacite)
	}

	Atelier o-- "1" Statut
	Atelier o-- "0:✻" Creneau
	Laboratoire o-- "0:✻" Atelier
	Utilisateur --|> Laboratoire
}
